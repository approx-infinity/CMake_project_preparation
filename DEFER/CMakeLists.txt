cmake_minimum_required(VERSION 3.19)
project(DeferMastery)
function(generate_plugin_summary)
    message(STATUS "===================")
    message(STATUS "DEFERED WORKAROUND")
    message(STATUS "===================")
    if(NOT REGISTERED_PLUGINS)
        message(STATUS "-> No plugins registered")
    else()
        foreach(plugins IN LISTS REGISTERED_PLUGINS)
            message(STATUS " --> Successfully linked: ${plugins}")
        endforeach()
    endif()
    message(STATUS "===================")
endfunction()
set(REGISTERED_PLUGINS "")
cmake_language(DEFER ID "summaryTask" CALL generate_plugin_summary)
cmake_language(DEFER ID "nagTask" CALL message WARNING "You didn't register any plugins!")
message(STATUS "Script started")
message(STATUS "[Adding Audio Plugins] ")
list(APPEND REGISTERED_PLUGINS "VideoPlugins")
cmake_language(DEFER GET_CALL_IDS my_queue)
message(STATUS "Current deferred queue ${my_queue}")
message(STATUS "Cancelling the 'nagTask' because plugins exist")
cmake_language(DEFER CANCEL_CALL "nagTask")
message(STATUS "Calling return to forcefully exit")
return()
