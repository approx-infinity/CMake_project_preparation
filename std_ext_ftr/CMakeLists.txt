cmake_minimum_required(VERSION 3.18)

set(TEST_STAND 11 14 17 20 23 26)
foreach(standard IN LISTS TEST_STAND)

    set(STANDARD_NAME CMAKE_CXX${standard}_STANDARD_COMPILE_OPTION)
    set(FEATURE_NAME cxx_std_${standard})
    message(STATUS "C++${standard}")

    message(STATUS "${STANDARD_NAME} = ${${STANDARD_NAME}}")
    message(STATUS "CMAKE_CXX_COMPILE_FEATURES = ${CMAKE_CXX_COMPILE_FEATURES}")
    message(STATUS "FEATURE_NAME = ${FEATURE_NAME}")
endforeach()
