set(SUPPORTED_AMD_ARCHS "VEGA90A;AMD_GFX942;AMD_GFX1100;AMD_GFX942")
set(SUPPORTED_AMD_GPUS  "MI100;MI300;RX7900;MI300A")
foreach(CURRENT_ARCH IN LISTS SUPPORTED_AMD_ARCHS)
    list(FIND SUPPORTED_AMD_ARCHS ${CURRENT_ARCH} FOUND_INDEX)
    list(GET SUPPORTED_AMD_GPUS ${FOUND_INDEX} MAPPED_GPU)
    message(STATUS "list(FIND) returned index : ${FOUND_INDEX}")
    message(STATUS "Mapped GPU result         : ${MAPPED_GPU}")
    message(STATUS "Arch: ${CURRENT_ARCH} - GPU: ${MAPPED_GPU}")
endforeach()
